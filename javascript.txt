function missingAngle(angle1, angle2) {
	let angle3 = 180 -(angle1+angle2)
	if (180==angle1+angle2+angle3) {
		if (angle3>90) {
			return 'obtuse'
		}
		else if (angle3<90) {
			return 'acute'
		}
		if (angle3==90) {
			return 'right'
		}
	}
}
